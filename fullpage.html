範例程式碼


<link href="https://cdnjs.cloudflare.com/ajax/libs/fullPage.js/3.1.2/fullpage.min.css" rel="stylesheet" />
<script src="https://cdnjs.cloudflare.com/ajax/libs/fullPage.js/3.1.2/fullpage.min.js"></script>

<div id="fullpage">
    <div class="section section1">第一部分</div>
    <div class="section section2">第二部分</div>
    <div class="section section3">第三部分</div>
    <div class="section section4">第四部分</div>
</div>

<script>
    new fullpage('#fullpage', {
        autoScrolling: true,
        navigation: true,
        scrollingSpeed: 700,
    });
</script>
