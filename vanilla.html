範例程式碼


            <script>
                const tiltBox = document.getElementById('tiltBox');
            
                tiltBox.addEventListener('mousemove', (e) => {
                    const { offsetWidth, offsetHeight } = tiltBox;
                    const { clientX, clientY } = e;
                    const x = (clientX / offsetWidth) - 0.5;
                    const y = (clientY / offsetHeight) - 0.5;
            
                    const tiltX = (y * 10).toFixed(2); // 最大傾斜角度
                    const tiltY = (x * -10).toFixed(2); // 最大傾斜角度
            
                    tiltBox.style.transform = `perspective(500px) rotateX(${tiltX}deg) rotateY(${tiltY}deg)`;
                });
            
                tiltBox.addEventListener('mouseleave', () => {
                    tiltBox.style.transform = 'perspective(500px) rotateX(0deg) rotateY(0deg)';
                });
            </script>
