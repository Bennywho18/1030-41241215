Anime.js 動畫展示

Anime.js 官網

範例程式碼


<script src="https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js"></script>

<div class="box"></div>

<script>
    anime({
        targets: '.box',
        translateX: 250,
        rotate: '1turn',
        duration: 2000,
        easing: 'easeInOutExpo',
        loop: true
    });
</script>
